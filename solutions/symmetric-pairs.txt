SELECT 
    F1.X AS X1,
    F1.Y AS Y1
FROM 
    Functions F1 JOIN
    Functions F2
    ON F1.X = F2.Y AND F2.X = F1.Y
WHERE F1.X < F1.Y
ORDER BY X1, Y1;
