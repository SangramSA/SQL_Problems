SELECT ROUND(AVG(LAT_N),4) AS MEDIAN FROM
​
(SELECT LAT_N,
      ROW_NUMBER() OVER (ORDER BY LAT_N ASC, id ASC) AS RowAsc,
      ROW_NUMBER() OVER (ORDER BY LAT_N DESC, ID DESC) AS RowDesc
   FROM STATION
    )data
WHERE
   RowAsc IN (RowDesc, RowDesc - 1, RowDesc + 1)
