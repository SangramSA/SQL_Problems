STUDENTS S JOIN PACKAGES SA
    ON 
        S.ID = SA.ID
),
FRIEND_SALARY AS (
    SELECT 
        F.ID AS FID, 
        F.FRIEND_ID, 
        SA.ID, 
        SA.SALARY 
    FROM 
        FRIENDS F JOIN PACKAGES SA
    ON 
        F.FRIEND_ID = SA.ID
)
SELECT 
    SS.NAME
FROM 
    STUDENTS_SALARY SS JOIN FRIEND_SALARY FS
ON 
    SS.SID=FS.FID
WHERE 
    SS.SALARY<FS.SALARY
ORDER BY 
    FS.SALARY;
