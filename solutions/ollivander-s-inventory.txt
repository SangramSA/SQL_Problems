SELECT W.id, WP.age, min(W.coins_needed), W.power
    FROM Wands W JOIN Wands_Property WP
        ON W.code = WP.code
    WHERE WP.is_evil = 0
    GROUP BY W.power
    ORDER BY W.power DESC, WP.age DESC;
