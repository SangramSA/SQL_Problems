    (SELECT 
        Start_Date 
    FROM
        Projects
    WHERE 
        Start_Date NOT IN (SELECT End_Date FROM Projects)) AS A
    INNER JOIN
SELECT 
    A.Start_Date, 
    min(B.END_DATE)
FROM
    (SELECT
        End_Date
    FROM 
        Projects
    WHERE END_DATE NOT IN (SELECT Start_Date FROM Projects)) AS B
WHERE 
    A.Start_Date<B.END_DATE
GROUP BY
    A.Start_Date
ORDER BY    
    min(B.END_DATE) -  A.Start_Date ASC, A.Start_Date ASC;
